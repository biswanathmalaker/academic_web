git clone https://github.com/biswanathmalaker/academic_web.git
cd academic_web
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt 


in academic_web/urls.py:

from django.contrib import admin
from django.urls import path,include
from django.conf import settings
from django.conf.urls.static import static

urlpatterns = [
    path('admin/', admin.site.urls),
    path("", include("home.urls")),
    path("papers/", include("papers.urls")),
    path("literature/", include("literature.urls")),
    ]

urlpatterns += static('/papers/pdf/', document_root=settings.BASE_DIR / 'papers/pdf')


in academic_web/settings.py:


INSTALLED_APPS = [
    'django.contrib.admin',
    'django.contrib.auth',
    'django.contrib.contenttypes',

# Create apps and then database.

python3 manage.py makemigrations papers literature home
python3 manage.py migrate

# run the project

python3 manage.py runserver 8080

Now you can add papers
there would be an unique id generated by the app for each paper like malaker-2024 , malaker-2024a etc.
